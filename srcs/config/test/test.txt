server {
	listen 127.0.0.1:80;
	server_name webserv.com;
	error_page /error-page.html;
	client_max_body_size 100M;

	location / {
		http_method GET POST DELETE;
		proxy_path http://127.0.0.1:8000;
		root /etc/www/html;
		autoindex on;
		index index.html;
		add_cgi_handler .cgi;
		is_uploadable on;
		upload_store /img;
	}
	location /img {
		http_method GET POST DELETE;
		proxy_path http://127.0.0.1:8000;
		root /etc/www/html;
		autoindex on;
		index index.html;
		add_cgi_handler .cgi;
		is_uploadable on;
		upload_store /img;
	}
}
